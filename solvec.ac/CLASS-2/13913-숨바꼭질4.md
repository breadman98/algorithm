```c++
#include <bits/stdc++.h>

using namespace std;
int n,k;
bool vis[100001];
int pre[100001];
queue<pair<int,int>> q; // position,time

void bfs(int st){
	q.push({st,0});
	vis[st] = true;
	
	while(!q.empty()){
        auto [cur,time] = q.front();
		q.pop();
		
		//if(cur<0 || cur>100000) continue;
		if(cur == k){
			cout<<time<<'\n';
			break;
		}
		
		if(cur+1<=100000 && !vis[cur+1]){
			vis[cur+1] = true;
			q.push({cur+1,time+1});
            pre[cur+1]=cur;
		}
		if(cur-1>=0 && !vis[cur-1]){
			vis[cur-1] = true;
			q.push({cur-1,time+1});
            pre[cur-1]=cur;
		}
		if(cur*2<=100000 && !vis[cur*2]){
			vis[cur*2] = true;
			q.push({cur*2,time+1});
            pre[cur*2]=cur;
		}
	}
}

void findPath(int start,int en){
    vector<int> path;
    int cur = en;
    while(cur!=start){
        path.push_back(cur);
        cur=pre[cur];
    }
    path.push_back(cur);
    reverse(path.begin(),path.end());
    for(auto x:path) cout<<x<<' ';

}

int main(){
	ios::sync_with_stdio(0);
	cin.tie(0);
	
	cin>>n>>k;
	bfs(n);
    findPath(n,k);
}
```
