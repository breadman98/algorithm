```
#include <bits/stdc++.h>

using namespace std;

int n,m;
int h[1000001];

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	
	cin>>n>>m;
	for(int i=0;i<n;i++){
		cin>>h[i];
	}
	sort(h,h+n);
	int bottom = 0;
	int top = h[n-1];
	
	int mid;
	while(bottom<top){
		mid = (bottom+top+1)/2;
		long sum = 0;
		for(auto height:h){
			if(height>mid) sum+=height-mid;
		}
	
		if(sum<m){
			top=mid-1;
		}
		else{
			bottom=mid;
		}
	
	}
	cout<<bottom;
}
```
