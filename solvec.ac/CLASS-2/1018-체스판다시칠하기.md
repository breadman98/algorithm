```c++
#include <bits/stdc++.h>

using namespace std;

int n,m;
int board[51][51];

int check(int n,int m){
	int cnt1=0;
	int cnt2=0;
	
	for(int i=n;i<=n+7;i++){
		for(int j=m;j<=m+7;j++){
			if((i+j)%2==board[i][j]) cnt1++;
			if((i+j+1)%2==board[i][j]) cnt2++;
		}
	}
	return min(cnt1,cnt2);
}


int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	
	cin>>n>>m;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			char c;
			cin>>c;
			if(c=='W') board[i][j] = 1;
			if(c=='B') board[i][j] = 0;
		}
	}
	
	int ans=10000;
	for(int i=0;i<n-7;i++){
		for(int j=0;j<m-7;j++){
//			if(ans>check(i,j)) ans = check(i,j);
			ans = min(ans,check(i,j));
		}
	}
	cout<<ans;
	
	
	return 0;
}
```
