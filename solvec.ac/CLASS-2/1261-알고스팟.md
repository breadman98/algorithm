```c++
#include <bits/stdc++.h>
using namespace std;

int m,n;
int dx[4]={1,0,-1,0};
int dy[4]={0,1,0,-1};
int dist[101][101];
string board[101];
deque<pair<int,int>> dq;

void bfs()
{
	dist[0][0]=0;
	dq.push_back({0,0});
	while(!dq.empty()){
		auto cur = dq.front(); dq.pop_front();
		for(int dir=0;dir<4;dir++){
			int nx = cur.first + dx[dir];
			int ny = cur.second + dy[dir];
			if(nx<0||ny<0||nx>=m||ny>=n) continue;
			if(dist[nx][ny]!=-1) continue;
			if(board[nx][ny]=='0'){
                dq.push_front({nx,ny});
                dist[nx][ny] = dist[cur.first][cur.second];
            }
            if(board[nx][ny]=='1'){
                dq.push_back({nx,ny});
                dist[nx][ny] = dist[cur.first][cur.second]+1;
            }
		}
	}
}


int main()
{
	cin>>n>>m;
	for(int i=0;i<m;i++) cin>>board[i];
	for(int i=0;i<m;i++){
		for(int j=0;j<n;j++){
			dist[i][j]=-1;
		}
	}	
	
	bfs();
	
	cout<<dist[m-1][n-1];
}
```
