```c++
#include <iostream>
#include <stack>

using namespace std;

int cursor = 0;

int main()
{
  stack<int> S;

  int k;
  cin>>k;
  
  int n;
  for(int i=0;i<k;i++){
    cin>>n;
    
    if(n==0) S.pop();
    else S.push(n);
  }

  int sum=0;
  while(!S.empty()){ // for문으로 하면 스택의 길이 잡기가 골아픔. 속편하게 while로 하자
    sum+=S.top(); // sum할 때 덧셈 해준애는 제외해야하니 pop
    S.pop();
  }
  cout<<sum;
  
}
```
