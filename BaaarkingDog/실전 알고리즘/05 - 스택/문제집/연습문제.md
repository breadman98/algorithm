```c++
#include <iostream>

using namespace std;

int arr[1000005];
int cursor = 0;


void push(int n){
  arr[cursor] = n;
  cursor++;
  
}

void pop(){
  cursor--;
}

int size(){
  return cursor-1;
}

int top(){
  return arr[cursor-1];
}

int main()
{
  ios::sync_with_stdio(false);
  cin.tie(NULL);

  int n;
  cin>>n;
  string str;

  for(int i=0;i<n;i++){
    cin>>str;

    if(str=="push"){
      int add;
      cin>>add;
      push(add);
    }
    else if(str=="pop"){
      if(cursor==0) cout<<-1<<'\n';
      else {
        cout<<top()<<'\n';
        pop();
      }
    }
    else if(str=="size"){ // 스택의 사이즈는 cursor현재 위치
      cout<<cursor<<'\n';
    }
    else if(str=="empty"){
      if(cursor==0) cout<<1<<'\n';
      else if(cursor>=1) cout<<0<<'\n';
    }
    else {
      if(cursor==0) cout<<-1<<'\n';
      else cout<<top()<<'\n';
    }
  }
}
```
