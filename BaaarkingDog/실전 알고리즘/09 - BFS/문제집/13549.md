```c++
#include <iostream>
#include <deque>

using namespace std;
int n,k;
bool visit[200001];
deque< pair<int,int> > Q;

void bfs(){
	Q.push_back({n,0});
	visit[n] = true;
	
	while(!Q.empty()){
		int cur = Q.front().first; 
		int time = Q.front().second;
		Q.pop_front();
		
		if(cur<0 || cur>100000) continue;
		if(cur == k){
			cout<<time;
			break;
		}
		if(!visit[cur*2]){
			visit[cur*2] = true;
			Q.push_front({cur*2,time});
		}
		if(!visit[cur+1]){
			visit[cur+1] = true;
			Q.push_back({cur+1,time+1});
		}
		if(!visit[cur-1]){
			visit[cur-1] = true;
			Q.push_back({cur-1,time+1});
		}
	}
}

int main(){
	ios::sync_with_stdio(0);
	cin.tie(0);
	
	cin>>n>>k;
	bfs();
}
```
