```c++
#include <bits/stdc++.h>

// end시간 정렬을 위해 end시간이 first에 오도록
#define end first
#define start second

using namespace std;

int n;
pair<int,int> v[100001]; // end, start

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	
	cin>>n;
	for(int i=0;i<n;i++) cin>>v[i].start>>v[i].end; // end시간 정렬을 위해 end시간이 first에 오도록 입력받음
	sort(v,v+n);
	
	int count=0;
	int endTime=0;
	for(int i=0;i<n;i++){
		if(endTime<=v[i].start){
			count++;
			endTime = v[i].end;
		}
	}
	
	cout<<count;
	return 0;
}
```
