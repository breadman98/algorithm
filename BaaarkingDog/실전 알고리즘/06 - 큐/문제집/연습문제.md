```c++
#include <iostream>

using namespace std;

int n;
int arr[100005];
int head = 0;
int tail = 0;
string str;

void push(int x){
  arr[tail] = x;
  tail++;
}

void pop(){
  head++;
}


int front(){
  return arr[head];
}

int back(){
  return arr[tail-1];
}


int main()
{
  ios::sync_with_stdio(false);
  cin.tie(NULL);

  cin>>n;

  while(n--){
    cin>>str;
    
    if(str=="push"){
      int add;
      cin>>add;
      push(add);
    }
    else if(str=="pop"){
      if(tail==head) cout<<-1<<'\n';
      else {
        cout<<front()<<'\n';
        pop();
      }
    }
    else if(str=="size"){
      cout<<tail-head<<'\n'; // 큐의 사이즈는 tail-head
    }else if(str=="empty"){
      if(tail==head) cout<<1<<'\n';
      else cout<<0<<'\n';
    }
    else if(str=="front"){
      if(tail==head) cout<<-1<<'\n';
      else
        cout<<front()<<'\n';
    }
    else if(str=="back"){
      if(tail==head) cout<<-1<<'\n';
      else cout<<back()<<'\n';
    }
  }
}
```
