	return false;

int Find(int u) {
	if (parent[u] == u) return u;
	return parent[u] = Find(parent[u]);
}

void Union(int u, int v) {
	u = Find(u), v = Find(v);
	if (u == v) return;
	parent[u] = v;
	Find(u);
}
